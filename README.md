# 言語 :: Igo

Support `zh-CN`, `ja-JP`

install
```cmd
gem install igo
```
# 使い方 :: Usage / Ja

```ruby
require 'igo'
require 'igo/ja'
cutted = Igo::Ja.cut "あー、合成は結合法則を満たすんでしたね"
#=> ["では", "、", "圏論", "の", "話", "を", "しましょ", "う", "か", "N", "この", "前", "は", "、圏", "について", "紹介 しました"]
cutted = Igo::Ja.cut "あー、合成は結合法則を満たすんでしたね", s: true
#=> "では 、 圏論 の 話 を しましょ う か N この 前 は 、圏 について 紹介しました"

j = Igo::Ja
```

# 用法 :: Usage / Zh

```ruby
require 'igo'
require 'igo/zh'
z = Igo::Zh

z.cut "全世界的无产阶级，联合起来！"
#=> ["全世界", "的", "無產階級", "，", "聯合", "起來", "！"]
z.cut "全世界的无产阶级，联合起来！", s: true
#=> "全世界 的 无产阶级 ！ 联合 起来 ！"

z.tag "全世界的无产阶级，联合起来！"
#=>  [["全世界", "n"], ["的", "uj"], ["无产阶级", "l"], ["！", "x"], ["联合", "v"], ["起来", "v"], ["！", "x"]]
z.tag "全世界的无产阶级，联合起来！", s: true
#=> "全世界_n 的_uj 无产阶级_l ，_x 联合_v 起来_v ！_x"

z.freq "全世界的无产阶级，联合起来！", 5
#=>  [["无产阶级", 7.39339993959], ["全世界", 6.80147590842], ["联合", 5.64979650728], ["起来", 3.96134044655]]
```
